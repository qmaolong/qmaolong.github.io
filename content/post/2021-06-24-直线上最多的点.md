func maxPoints(points [][]int) int {
	l := len(points)

	//åˆ¤æ–­3ä¸ªç‚¹æ˜¯å¦ä¸ºä¸€æ¡ç›´çº¿
	var ifLine = func(p1, p2, p3 []int) bool {
		a1 := p1[0] - p2[0]
		a2 := p1[1] - p2[1]
		b1 := p2[0] - p3[0]
		b2 := p2[1] - p3[1]

		if a1 == 0 || b1 == 0 {
			return a1 == b1 && p1[0] == p2[0] && p1[0] == p3[0]
		}
		if a2 == 0 || b2 == 0 {
			return a2 == b2 && p1[1] == p2[1] && p1[1] == p3[1]
		}
		return a1*b2 == a2*b1
	}

	arr := make([][]*int, l)
	for i := range arr {
		arr[i] = make([]*int, l)
	}

	max := 1
	for i := range points {
		for j := i + 1; j < l; j++ {
			if arr[i][j] != nil {
				continue
			}
			tCount := 2
			arr[i][j] = &tCount
			for k := j + 1; k < l; k++ {
				if ifLine(points[i], points[j], points[k]) {
					tCount++
					arr[i][k] = &tCount
					arr[j][k] = &tCount
				}
			}
			if tCount > max {
				max = tCount
			}
		}
	}

	return max
}
